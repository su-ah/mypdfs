we want $k$-forms to be sth we integrate over $k$-dim submflds.
(in)formally, this is a map:
\begin{equation*}
	\omega:\underbrace{T_{p}M\times \cdots \times T_{p}M}_{k \trm{ times}}\gto\bR
\end{equation*}
then, for $F:U\subseteq\bR^{k}\gto M$, we can define:
\begin{equation*}
	\int_{F(U)}\omega
	\ = \ \int_{U}\omega_{F(x)}(F_{*}\p_{1},\dots,F_{*}\p_{k})
	\di x_{1}\cdots\d x_{k}
\end{equation*}

we \textit{want} invariance under reparametrization $G:U\gto V$ diffeo, that is:
\begin{equation*}
	\ = \ \int_{G(U)}\omega_{F(G^{-1}(y))}
	\left((F\circ G^{-1})_{*}\p_{1},\dots,(F\circ G^{-1})_{*}\p_{k}\right)
	\di y_{1}\cdots\d y_{k}
\end{equation*}
which, by change of variables from 257, is given as:
\begin{equation*}
	\ = \ \pm\int_{U}\omega_{F(x)}
	\left((F\circ G^{-1})_{*}\p_{1},\dots,(F\circ G^{-1})_{*}\p_{k}\right)
	\abs{\det TG^{-1}}\di x_{1}\cdots\d x_{k}
\end{equation*}

so what do we acc need? recall the following fact: if $V$ is $k$-dim'l, then
given any map $\omega:V\times\cdots\times V\gto\bR$ which is multilinear and
skew-symmetric, then we have that
$\omega(v_{1},\dots,v_{k})=C\cdot\det[v_{1} \ \cdots \ v_{k}]$.
thus, if $\omega:V^{k}\gto\bR$ for $n$-dim'l $V$ is multilinear and alternating,
its restriction to \textit{any} $k$-dim subspace behaves like a det.

denoting these as alternating (or skew-sym) $k$-tensors,
our vague defn is that a $k$-form $\omega\in\Omega^{k}(M)$ is a smooth choice of
alt $k$-tensors $\omega_{p}:T_{p}M\times\cdots\times T_{p}M\gto\bR$.

\begin{defn}[title=Functional]
	a $k$\textbf{-form} is an alternating multilinear map:
	$\omega:\underbrace{\frk{X}(M)\times\cdots\times\frk{X}(M)}_{k \trm{ times}}\gto C^{\infty}(M)$.
\end{defn}
for instance, $\omega\in\Omega^{n}(\bR^{n})$ always looks like:
\begin{equation*}
	\omega(X_{1},\dots,X_{n})(p) \ = \ f(p)\det[(X_{1})_{p} \ \cdots \ (X_{n})_{p}]
\end{equation*}
for some $f\in C^{\infty}(\bR^{n})$.
what do alt $k$-tensors look like in general?

\begin{defn}
	let $I=\set{i_{1}<\dots<i_{k}}\subseteq\set{1,\dots,n}$.
	define $\d x_{I}=\d x_{i_{1}}\wedge\cdots\wedge \d x_{i_{k}}$
	to be the alt $k$-tensor which satisfies:
	\begin{equation*}
		\d x_{I}(\p_{i_{1}},\dots,\p_{i_{k}}) = 1 \qquad
		\d x_{I}(p_{j},\trm{ anything else }) = 0
	\end{equation*}
	where $j\notin I$. note this uniquely defns an alt $k$-tensor.
\end{defn}
here, the $\wedge$ is simply a symbol (for now).
intepretation: take $v_{1},\dots,v_{k}$ and compute
$\det[\pi_{I}v_{1} \ \cdots \ \pi_{I}v_{k}]$, where $\pi_{I}$ projects
to $\ang{\p_{i_{1}},\dots,\p_{i_{k}}}$.

\begin{lm}[title=Coordinate-wise]
	every alt $k$-tensor $\omega:(\bR^{n})^{k}\gto\bR$ can be expressed
	uniquely as:
	\begin{equation*}
		\omega \ = \ \sum_{\substack{I\subseteq\set{1,\dots,n}\\\abs{I}=k}}\omega_{I}\di x_{I} \qquad
		\omega_{I}\in\bR
	\end{equation*} \vspace{-0.3in}
\end{lm} \

\begin{pf}
	note $\omega_{I}=\omega(\p_{i_{1}},\dots,\p_{i_{k}})$,
	check that $\omega(v_{1},\dots,v_{k})$ determined by these. (what?)
\end{pf}

\begin{prop}
	any $k$-form on $U\subseteq\bR^{n}$ is given by:
	\begin{equation*}
		\omega \ = \ \sum_{\substack{I\subseteq\set{1,\dots,n}\\\abs{I}=k}}
		f_{I}(p)\di x_{I}
	\end{equation*}
	for $f_{I}:U\gto\bR$ smooth.
	any $k$-form on $M$ can be written this way on charts.
	in particular, $\omega(X_{1},\dots,X_{k})(p)=\omega_{p}((X_{1})_{p},\dots,(X_{k})_{p})$
	for some well-defined alt $k$-tensor $\omega_{p}$.
\end{prop}

we start w the wedge product.
this is an operation $\Omega^{k}(M)\times\Omega^{\ell}(M)\gto\Omega^{k+\ell}(M)$.
as warmup, lets look at the case where $k=\ell=1$.

let $\alpha,\beta\in\Omega^{1}(M)$.
does $\alpha\wedge\beta(v_{1},v_{2})=\alpha(v_{1})\beta(v_{2})$ work?
not necessarily, since it may not be alternating.
we can instead force this by instead setting:
\begin{equation*}
	\alpha\wedge\beta(v_{1},v_{2}) \ := \ \alpha(v_{1})\beta(v_{2})-\beta(v_{1})-\alpha(v_{2})
\end{equation*}
for instance, on $\bR^{2}$ with $\alpha=\d x,\beta=\d y$, sps $v_{i}=(x_{i},y_{i})$. then:
\begin{equation*}
	\alpha\wedge\beta(v_{1},v_{2}) \ = \ x_{1}y_{2}-x_{2}y_{1}
	\ = \ \det\begin{bmatrix} x_{1} & x_{2} \\ y_{1} & y_{2} \end{bmatrix}
	\ = \ \d x\wedge\d y(v_{1},v_{2})
\end{equation*}
more generally, on $\bR^{n}$ with $\alpha=\d x_{1},\beta=\d x_{2}$, we have that
$\alpha\wedge\beta(\p_{i},\p_{j})=0$ unless $i,j=1,2$ in some order,
$\alpha\wedge\beta(\p_{1},\p_{2})=1$, and $\alpha\wedge\beta(\p_{2},\p_{1})=-1$.

\begin{defn}
	sps $\alpha\in\Omega^{k}(M),\beta\in\Omega^{\ell}(M)$.
then the \textbf{wedge product} $\alpha\wedge\beta\in\Omega^{k+\ell}(M)$ is: \vspace{-0.1in}
	\begin{align*}
		\alpha\wedge\beta(v_{1},\dots,v_{k+\ell})
		& \ = \ \frac{1}{k!\ell!} \sum_{\sigma\in S_{k+\ell}} \trm{sign}(\sigma)
		\alpha(v_{\sigma(1)},\dots,v_{\sigma(k)})
		\beta(v_{\sigma(k+1),\dots,v_{\sigma(k+\ell)}}) \\
		& \ = \ \sum_{\sigma\in\Sigma_{k,\ell}}\trm{sign}(\sigma)
		\alpha(v_{\sigma(1)},\dots,v_{\sigma(k)})
		\beta(v_{\sigma(k+1)},\dots,v_{\sigma(k+\ell)})
	\end{align*}
	where $\Sigma_{k,\ell}$ represents the ``$(k,\ell)$ shuffles", that is,
	permutations which preserve the independent ordering of $[k]$ and
	$[k+\ell]\sm[k]$.
\end{defn}
in particular, $\d x_{I}=\d x_{i_{1}}\wedge\cdots\wedge\d x_{i_{k}}$ for
$i_{1}<\cdots<i_{k}$.
that is, our earlier ``symbol" was in fact the right thing.

some properties of the wedge product:

\end{enumerate}

ok moving on to pullbacks of forms.
\begin{defn}
	given $F:M\gto N$, $\alpha\in\Omega^{k}(N)$, we define:
	\begin{equation*}
		F^{*}\alpha(v_{1},\dots,v_{k}) \ = \ \alpha(F_{*}v_{1},\dots,F_{*}v_{k})
	\end{equation*}
\end{defn}

properties:
\begin{enumerate}[i)]
	\item is in fact a diff form (varies smoothly)
	\item preserves wedge: $F^{*}(\alpha\wedge\beta) \ = \ F^{*}\alpha\wedge F^{*}\beta$
	\item preserves $\d$: $F^{*}(\d\alpha) \ = \ \d F^{*}\alpha$
\end{enumerate}

\begin{pf}\vspace{-0.2in}
	\begin{enumerate}
		\item pointwise, (ii) holds:
			\begin{align*}
				F^{*}(\alpha\wedge\beta)(v_{1},\dots,v_{k})
				& \ = \ (\alpha\wedge\beta)(F_{*}v_{1},\dots,F_{*}v_{k}) \\
				& \ = \ \sum_{\sigma \trm{ shuffles}}\alpha(F_{*}v_{\sigma(1)},\dots,F_{*}v_{\sigma(k)})\beta(\cdots) \\
				& \ = \ F^{*}\alpha\wedge F^{*}\beta(v_{1},\dots,v_{k})
			\end{align*}
		\item $\d,\wedge,F^{*}$ are all $\bR$-lin maps + local, so
			we only have to show this for $f_{0}\d f_{1}\wedge\cdots\wedge\d f_{k}$
		\item by step 1:
			\begin{equation*}
				F^{*}(f_{0}\d f_{1}\wedge\cdots\wedge\d f_{k})
				\ = \ F^{*}f_{0}F^{*}\d f_{1}\wedge\cdots\wedge F^{*}\d f_{k}
			\end{equation*}
			we also know pullbacks of 1-forms are 1-forms, so we get (i).
			we also also know $F^{*}(\d f)=\d F^{*}f$, so we get (iii).
	\end{enumerate}
\end{pf}

in particular, we see from last time that for $F:\bR^{k}\gto\bR^{k}$,
every $k$-form $\omega$ is:
\begin{equation*}
	\omega \ = \ f\d x_{1}\wedge\cdots\wedge\d x_{k}
\end{equation*}
for some $f$. thus:

	F^{*}\omega(\p_{1},\dots,\p_{k})

	& \ = \ \omega(\p_{1}F,\dots,\p_{k}F) \\
	& \ = \ \det(TF)\omega(\p_{1},\dots,\p_{k})
\end{align*}
ie, $F^{*}\omega\big\rvert_{p} \ = \ \det(T_{p}F)\omega\big\rvert_{F(p)}$.
