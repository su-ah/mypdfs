\begin{defn}
	a set $S^{k}\subseteq M^{m}$ is a \textbf{submfld} if $\fa p\in S$,
	there is a chart $(U_{p},\vphi_{p})$ of $M$ arnd $p$ s.t.:
	\begin{equation*}
		\vphi_{p}(U_{p}\cap S) \ = \ \vphi_{p}(U_{p})\cap\bR^{k}
	\end{equation*}
	where $\bR^{k}\subseteq\bR^{m}$ is the first $k$ coordinates.
	such a chart is called a \textbf{submfld chart}.
\end{defn}
intuitively, $S\subseteq M$ can be represented purely with the first $k$ coords
for any local coords on $S$. thus, its a $k$-dim submfld w local coords given by
restricting to the first $k$ coords.

\begin{prop}
	the set $\set{(U_{p}\cap S,\vphi_{p}\big\rvert_{U_{p}\cap S}):p\in S}$ is
	a [smooth] atlas on $S$, such that $S$ is a mfld and $\iota:S\ito M$ is smooth.
	note each $\vphi_{p}$ is of the form $U_{p}\cap S\gto\bR^{k}$.
\end{prop}
can alternatively treat $\vphi_{p}$ as composition of proj w $\vphi_{p}$ from $M$.

\begin{pf}[source=Primary Source Material]
	we need to prove the following:
	\begin{enumerate}[(1)]
		\item charts cvr $S$ and are compat

			exercise. point is: restriction of smooth is smooth, and
			$U\cap\bR^{k}$ open in $\bR^{k}$ for open $U\subseteq\bR^{n}$.

		\item hausdorff

			given $p,q\in S$, choose $(U,\vphi),(V,\psi)$ submfld charts at $p,q$.
			then $p\in U',q\in V'$ disj open in $M$, so $\vphi\rvert_{U\cap U'},\psi\rvert_{V\cap V'}$
			submfld charts w $S\cap U\cap U'$ disj from $S\cap V\cap V'$.

		\item $\ex$ ctbl atlas

			$\fa p\in S$, choose $V_{i}$ from a ctbl basis of $M$ s.t.
			$p\in V_{i}\subseteq U_{p}$. then $\vphi_{i}:V_{i}\gto\bR^{n}$
			gives a submfld chart dep on $i$.

		\item $\iota$ is smooth

			fix $p\in S$ and pick a chart $U_{p}\cap S$. then $\iota$ smooth at $p$ since:
			\begin{equation*}
				\vphi_{p}\circ\iota\circ(\vphi_{p}\big\rvert_{S})^{-1} :
				\vphi(U_{p})\cap\bR^{k}\gto\vphi(U_{p})
			\end{equation*}
			is given by the id.
	\end{enumerate}
\end{pf} \

\begin{prop}
	let $S^{k}\subseteq M^{m}$.
	then $V\subseteq S$ open iff $V=U\cap S$ for some open $U\subseteq M$.
\end{prop}
in other words, topo on $S$ is precisely the subspace topo from $M$.

\begin{pf}[source=Primary Source Material]
	$(\Longleftarrow)$ let $(V,\psi)$ be a submfld chart of $S$. then:
	\begin{equation*}
		\psi'((V\cap S)\cap(U\cap S)) \ = \ (\pi\circ\psi)(V\cap S\cap U)
		\ = \ \pi(\psi(V)\cap\psi(U)\cap\bR^{k})
	\end{equation*}
	since $\psi(V)\cap\psi(U)\cap\bR^{k}$ open in $\bR^{k}$, we're done.

	$(\implies)$ sps $U'$ open in $S$. let:
	\begin{equation*}
		U=\bigcup_{V}\psi^{-1}(\psi'(U'\cap V)\times\bR^{m-k}) \ \subseteq \ M
	\end{equation*}
	note that this satisfies $U\cap S=U'$ (stare at it).
	furthermore, it is a union of open sets, since $U'$ open in $S$ implies
	$U'\cap V$ open in $S$, so each set is open in $M$.
\end{pf} \
