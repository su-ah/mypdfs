\subsection{The Jacobson Radical}
\setcounter{subsection}{1}

\begin{exr}[title=secretly Lemma 4.1 part a)]
	Let $R$ be a ring.
	Define the Jacobson radical $J(R)$ as the intersection of all maximal left ideals.
	Prove that $J(R)$ is, in fact, a two-sided ideal.
	(As a lemma, prove that an element $x\in R$ belongs to the Jacobson radical if and only if $1-axb$ is invertible for all $a,b\in R$.)
\end{exr}

\end{exr}


\nq
\begin{exr}[title=secretly Theorem 4.14 part a)]
	A ring is said to be \textbf{semiprimitive} if $J(R)=\{0\}$.
	Prove that if $R$ is a semisimple ring, then
	$R$ is artinian and semiprimitive.
\end{exr} \

\begin{pf}
	Suppose $R$ is semisimple. Then, there exists some left ideal $I$ such that:
	\begin{equation*}
		R \ = \ J(R)\oplus I
	\end{equation*}
	If $J(R)\neq(0)$, then $I\neq R$, and is thus contained in some maximal ideal $I'$.
	But then $I'\nsubseteq J(R)$, a contradiction.
	Thus, $R$ must be semiprimitive.
	Since a ring is semisimple if and only if it is noetherian and artinian,
	we are done.
\end{pf}
